#define StdArray(T, N)
Macro alias for a StdArray(T, N) containing up to N Ts.
#define StdArrayIterator(T, N)
macro alias for the concrete type for an iterator into the mutable iteration of a StdArray(T, N) containing up to N Ts
#define StdArrayConstIterator(T, N)
macro alias for the concrete type for an iterator into the const iteration of a StdArray(T, N) containing up to N Ts
#define StdArrayIdentifier(T, N, Identifier)
macro alias for an identifier (type, function, etc) associated with a StdArray(T, N) containing up to N Ts
#define std_array_new(T, N)
Creates a new StdArray(T, N) with defaulted associated functions.
#define std_array_new_with_allocator(T, N, allocator)
Creates a new StdArray(T, N) with defaulted associated functions.
#define std_array_new_with_collection_data(T, N, collection_data_ptr)
Creates a new StdArray(T, N) with provided associated functions.
#define std_array_new_with_allocator_and_collection_data(T, N, allocator, collection_data_ptr)
Creates a new StdArray(T, N) with provided associated functions.
#define std_array_clone(self)
Clones the given StdArray(T, N)
#define std_array_at_mut(self, index)
Returns a mutable reference to the element at the given index into the given StdArray(T, N)
#define std_array_at(self, index)
Returns a const reference to the element at the given index into the given StdArray(T, N)
#define std_array_front_mut(self)
Returns a mutable reference to the first element in the given StdArray(T, N)
#define std_array_front(self)
Returns a const reference to the first element in the given StdArray(T, N)
#define std_array_back_mut(self)
Returns a mutable reference to the last element in the given StdArray(T, N)
#define std_array_back(self)
Returns a const reference to the last element in the given StdArray(T, N)
#define std_array_data_mut(self)
Returns a mutable reference to the internal array in the given StdArray(T, N)
#define std_array_data(self)
Returns a const reference to the internal array in the given StdArray(T, N)
#define std_array_is_empty(self)
Returns whether the given StdArray(T, N) is empty.
#define std_array_is_full(self)
Returns whether the given StdArray(T, N) is full.
#define std_array_size(self)
Returns the current size of the given StdArray(T, N)
#define std_array_capacity(self)
Returns the capacity of the given StdArray(T, N) (ie the N)
#define std_array_resize(self, new_size)
Resizes the given StdArray(T, N) to new_size
#define std_array_clear(self)
Resets the given StdArray(T, N) to a size of zero, destroying any active elements in the array.
#define std_array_push_back(self, element)
Pushes a new element into the given StdArray(T, N), at the end.
#define std_array_pop_back(self)
Removes the last element from the given StdArray(T, N) and returns it.
#define std_array_insert(self, element, index)
Inserts the given element at the given index in the StdArray(T, N)
#define std_array_erase(self, index)
Erases the element at index from the StdArray(T, N)
#define std_array_erase_n(self, index, num_elements)
Erases num_elements elements from the StdArray(T, N), starting at index
#define std_array_free(self)
Frees the given StdArray(T, N), destroying its elements.
#define std_array_begin(self)
Returns a StdRandomAccessIterator into the mutable iteration of the given StdArray(T, N), starting at the beginning of the iteration (pointing at the beginning of the array)
#define std_array_end(self)
Returns a StdRandomAccessIterator into the mutable iteration of the given StdArray(T, N), starting at the end of the iteration (pointing at the end of the array)
#define std_array_rbegin(self)
Returns a StdRandomAccessIterator into the mutable iteration of the given StdArray(T, N), starting at the beginning of the reversed iteration (pointing at the end of the array)
#define std_array_rend(self)
Returns a StdRandomAccessIterator into the mutable iteration of the given StdArray(T, N), starting at the end of the reversed iteration (pointing at the beginning of the array)
#define std_array_iterator_equals(first, second)
Returns whether the given pair of iterators are equal (they belong to the same collection and point to the same element), IE: if first == second
#define std_array_cbegin(self)

#define std_array_cend(self)
Returns a StdRandomAccessIterator into the const iteration of the given StdArray(T, N), starting at the end of the iteration (pointing at the end of the array)
#define std_array_crbegin(self)
Returns a StdRandomAccessIterator into the const iteration of the given StdArray(T, N), starting at the beginning of the reversed iteration (pointing at the end of the array)
#define std_array_crend(self)
Returns a StdRandomAccessIterator into the const iteration of the given StdArray(T, N), starting at the end of the reversed iteration (pointing at the beginning of the array)
